					superregion = south_america_superregion

					superregion = oceania_superregion

				}
				NOT = {
					has_province_modifier = close_area_5_modifier
				}
				owner = {
					ai = no
				}
			}

			}

	}

		trigger = {
			has_province_modifier = close_area_5_modifier
		}
		remove_province_modifier = close_area_5_modifier
	}
